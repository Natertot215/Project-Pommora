**G.3 — Markdown auto-pair** (new file: [`CoreEditor/src/extensions/markdown-autopair.ts`](External/MarkdownEditor/CoreEditor/src/extensions/markdown-autopair.ts))

```ts
import { EditorView } from "@codemirror/view";
import { EditorSelection, Extension } from "@codemirror/state";
import { syntaxTree } from "@codemirror/language";

const PAIRS: Record<string, [string, string]> = {
  "*": ["**", "**"],
  "_": ["__", "__"],
  "[": ["[[", "]]"],
  "`": ["``", "``"],
};

export const markdownAutoPair: Extension = EditorView.inputHandler.of(
  (view, from, to, text) => {
    const pair = PAIRS[text];
    if (!pair) return false;
    const state = view.state;
    // Skip inside code blocks (don't auto-pair markdown chars inside ``` ``` blocks).
    const nodeName = syntaxTree(state).resolveInner(from, -1).name;
    if (/Code|FencedCode|InlineCode/.test(nodeName)) return false;

    const range = state.selection.main;

    // Non-empty selection: wrap.
    if (!range.empty) {
      const sel = state.sliceDoc(range.from, range.to);
      view.dispatch({
        changes: { from: range.from, to: range.to, insert: pair[0] + sel + pair[1] },
        selection: EditorSelection.range(
          range.from + pair[0].length,
          range.to + pair[0].length
        ),
        userEvent: "input.pair",
      });
      return true;
    }

    // Empty selection: only fire when the PREVIOUS char equals the typed char
    // (so first `*` types normally, second `*` triggers the pair).
    const prev = state.sliceDoc(Math.max(0, from - 1), from);
    if (prev !== text) return false;

    view.dispatch({
      changes: { from, to, insert: text + pair[1] },
      selection: EditorSelection.cursor(from + 1),
      userEvent: "input.pair",
    });
    return true;
  }
);
```

Wire in [`base.ts`](External/MarkdownEditor/CoreEditor/src/extensions/base.ts) — append `markdownAutoPair` after `closeBrackets()`. `closeBrackets` still handles single-char `()`, `[]`, `{}`, `""`, `''`, single `` ` ``. Our handler fires on the SECOND char of `**`, `__`, `[[`, `` `` `` — wikilink + bold + italic-via-underscore + double-backtick inline code.

**G.4 — Tighten heading→body spacing** (theme CSS in [`themes/base.ts`](External/MarkdownEditor/CoreEditor/src/ui/themes/base.ts))

Add to the theme rules:
```ts
".cm-line:has(.cm-header)": {
  paddingTop: "0.25em",
  paddingBottom: "0.1em",
  lineHeight: "1.2",
},
".cm-header": {
  lineHeight: "1.2",
},
```

This reduces the gap between a heading line and the following body line. The default 1.55-1.6 line-height around headings creates the large vertical gap Nathan flagged. (If `:has()` selector isn't supported in the WKWebView Safari version, fall back to a `cm-headerLine` decoration class added by a small `ViewPlugin` — but WKWebView on macOS 14+ supports `:has()`.)
